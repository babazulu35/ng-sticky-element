import { Directive, AfterViewInit, ViewContainerRef, Inject, Input,ElementRef } from '@angular/core';
import { Observable } from 'rxjs';

@Directive({
  selector: '[sticky]'
})
export class StickyDirective {
  
  /**
   * Directive set options
   */
  directiveOptions = {

    pinToParent: false

  }

  scrolListener:Observable<any>;
  
  /**
   * Element Query Selector
   */
  querySelectorName:ElementRef;

  /**
   * Parent from stacked Element
   */
  pinnedTo;

  

  /**
  * Stick the element to Parent otherwise is pinned to document scroll
  */

  @Input('sticky-pin-to-parent') set pinToParent(pinActive:string) {
    console.log("PinActive",pinActive);
    if(pinActive == '' || pinActive == undefined ) {
        this.directiveOptions.pinToParent = true;
    }
    else {
      this.directiveOptions.pinToParent = false ;
    }
  }
  
  
  constructor( 
    @Inject(ViewContainerRef) private viewContainer: ViewContainerRef,
    private el:ElementRef
  ) { }
  
  
  
  
  ngAfterViewInit() {
    
    this.directiveOptions.pinToParent ? this.pinnedTo = this.el.nativeElement.parentElement : this.pinnedTo = document;

    this.scrolListener = Observable.fromEvent(this.pinnedTo,"scroll").map(e => {return e });
    this.scrolListener.subscribe(result => {
      console.log("Scroll Listener",result);
    })
  }

}
